 /**
     * this method is used for post
     */
    private void callToAPIPost() {
        HashMap<String, String> headers = new HashMap<String, String>();
        headers.put("type", "json");
        DemoHttpHandler demoHttpHandler = new DemoHttpHandler(ActivityLogin.this, this, "Please wait....", 1);
        RequestParams params = new RequestParams();
        //grant_type=password&username=preeti@yopmail.com&password=Qwerty1@
        params.add("grant_type", "password");
        params.add("username", username);
        params.add("password", password);
        HttpClients.post(context, NetworkConstants.URL.Login, params, demoHttpHandler, headers);
    }

    /**
     * this method is used for get
     */
    private void callToAPIGet() {
        HashMap<String, String> headers = new HashMap<String, String>();
        headers.put("type", "json");
        DemoHttpHandler demoHttpHandler = new DemoHttpHandler(ActivityLogin.this, this, "Please wait...!", 2);
        HttpClients.get(context, "Your URL", null, demoHttpHandler, headers);
    }
